def name_repair(name):
    name = name.replace('[','').replace(']','')
    ## タイトルに使えない文字の排除
    for i in '\\' , '\"' , '\'' , '/' , ':' , '*' , '?' , '<' , '>' , '|' , '[' , ']' , ' ' , '　':
        name = name.replace(i,'_')
    while '__' in name:
        name = name.replace('__','_')
    while '_-_' in name:
        name = name.replace('_-_','-')
    return name
